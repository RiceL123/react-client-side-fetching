Client-side fetching in React

Agenda

Assumptions: You have a basic grasp of React's useEffect

  • classic fetch in useEffect
  • Suspense + use
  • Libraries like SWR and Tanstack Query (React Query)

fetch in Effects (the classic approach)


import { useState, useEffect } from "react";

const CAT_FACT_URL = "https://catfact.ninja/fact";

export default function App() {
  const [isLoading, setIsLoading] = useState(false);
  const [catFact, setCatFact] = useState(null);

  useEffect(() => {
    async function fetchCatFacts() {
      setIsLoading(true);
      const response = await fetch(CAT_FACT_URL);
      const json = await response.json();
      setCatFact(json["fact"]);
      setIsLoading(false);
    }

    fetchCatFacts();
  }, []); // empty dependency array to run once on mount

  if (isLoading) return <p>Loading...</p>;
  return <p>{catFact}</p>;
}

useEffect with Custom Hook


import { useState, useEffect } from "react";

const CAT_FACT_URL = "https://catfact.ninja/fact";

function useCatFact() { // 'use' prefix for custom hook
  const [isLoading, setIsLoading] = useState(false);
  const [catFact, setCatFact] = useState(null);

  useEffect(() => {
    async function fetchCatFacts() {
      setIsLoading(true);
      const response = await fetch(CAT_FACT_URL);
      const json = await response.json();
      setCatFact(json["fact"]);
      setIsLoading(false);
    }

    fetchCatFacts();
  }, []);

  return { isLoading, catFact };
}

export default function App() {
  const { catFact, isLoading } = useCatFact();

  if (isLoading) return <p>Loading...</p>;
  return <p>{catFact}</p>;
}

This works

But it is not the best solution we have in 2025

  • The manual management of state is imperative making its harder to reason about and maintain
  • It is less performant because it has to wait for the component to render before fetching

So let's update it to use modern React v19

Note: If you see a double fetch its due to StrictMode which helps catch errors during development. In production only one fetch will be called

Suspense + use


import { Suspense, use } from "react";

const CAT_FACT_URL = "https://catfact.ninja/fact";

function CatFact({ catFactPromise }) {
  const catFact = use(catFactPromise); // suspends until promise resolves
  return <p>{catFact}</p>;
}

export default function App() {
  const catFactPromise = fetch(CAT_FACT_URL)
    .then((res) => res.json())
    .then((json) => json.fact);

  return (
    <Suspense fallback={<p>Loading...</p>}>
      <CatFact catFactPromise={catFactPromise} />
    </Suspense>
  );
}

Suspense + use

  • available since React v19+ (dec 2024)
  • React.use does not support promises created in render
  • Suspense is designed for resources consumed directly during render so it cannot detect fetches inside an Effect or event handler
External Libraries

SWR (stale-while-revalidate)


import useSWR from "swr";

const CAT_FACT_URL = "https://catfact.ninja/fact";

const fetcher = (url) => fetch(url).then((res) => res.json());

export default function App() {
  const { data, isLoading } = useSWR(CAT_FACT_URL, fetcher);

  if (isLoading) return <p>Loading...</p>;
  return <p>{data.fact}</p>;
}

SWR (stale-while-revalidate) + Suspense


import { Suspense } from "react";
import useSWR from "swr";

const CAT_FACT_URL = "https://catfact.ninja/fact";

const fetcher = (url) => fetch(url).then((res) => res.json());

function CatFact() {
  const { data } = useSWR(CAT_FACT_URL, fetcher, { suspense: true });
  return <p>{data.fact}</p>;
}

export default function App() {
  return (
    <Suspense fallback={<p>Loading...</p>}>
      <CatFact />
    </Suspense>
  );
}

SWR (stale-while-revalidate)

  • ~8kb
  • Simple and avoids boiler plate
  • Manages state for you and provides loading, error and success states
  • Flexible for how you want to design your code (with or without Suspense)
  • Performant as it caches queries and has options for mutations and more

TanStack Query (formerly React Query)


import { useQuery } from "@tanstack/react-query";

const CAT_FACT_URL = "https://catfact.ninja/fact";

function useCatFact() {
  return useQuery({
    queryKey: ["cat-fact"],
    queryFn: async () => {
      const response = await fetch(CAT_FACT_URL);
      return await response.json();
    },
  });
}

export default function App() {
  const { data, isLoading } = useCatFact();

  if (isLoading) return <p>Loading...</p>;
  return <p>{data.fact}</p>;
}

import { Suspense } from "react";
import { useSuspenseQuery } from "@tanstack/react-query";

const CAT_FACT_URL = "https://catfact.ninja/fact";

function useCatFact() {

function CatFact() {
  const { data } = useCatFact();
  return <p>{data.fact}</p>;
}

export default function App() {
  return (
    <Suspense fallback={<p>Loading...</p>}>
      <CatFact />
    </Suspense>

TanStack Query

  • ~20kb
  • Comprehensive set of features
  • Manages state for you and provides loading, error and success states
  • Flexible for how you want to design your code (with or without Suspense)
  • Performant with caching, retry logic, background updates and more out the box

Conclusion

  • Stop fetching in useEffect - it's less maintainable and less performant
  • Use modern React patterns or popular libraries (or both) instead
